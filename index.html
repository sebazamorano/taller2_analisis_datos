from sklearn.linear_model import Ridge, Lasso, ElasticNet

alphas = [0.5, 2, 10, 0.01, 30, 40]

best_model = None
best_mse = float('inf')

for alpha in alphas:
    ridge_reg = Ridge(alpha=alpha, solver="cholesky")
    ridge_reg.fit(X_train, y_train)
    y_pred = ridge_reg.predict(X_test)
    mse = mean_squared_error(y_test, y_pred)
    rmse = root_mean_squared_error(y_test, y_pred)

    print(f"Ridge - Alpha: {alpha}, MSE: {mse}, RMSE: {rmse}")

    lasso_reg = Lasso(alpha=alpha)
    lasso_reg.fit(X_train, y_train)
    y_pred = lasso_reg.predict(X_test)
    mse = mean_squared_error(y_test, y_pred)
    rmse = root_mean_squared_error(y_test, y_pred)

    print(f"Lasso - Alpha: {alpha}, MSE: {mse}, RMSE: {rmse}")

    ratios = [0.5, 0.25, 0.75]

    for ratio in ratios:
      elastic_net = ElasticNet(alpha=alpha, l1_ratio=ratio, max_iter=1000)
      elastic_net.fit(X_train, y_train)
      y_pred = elastic_net.predict(X_test)
      mse = mean_squared_error(y_test, y_pred)
      rmse = root_mean_squared_error(y_test, y_pred)

      print(f"ElasticNet - Alpha: {alpha}, Ratio: {ratio}, MSE: {mse}, RMSE: {rmse}")

      if mse < best_mse:
        best_mse = mse
        best_model = (ridge_reg, lasso_reg, elastic_net, mse)
